{% block extra_js %}
<script>
// Gestion du lettrage
const ecritureChecks = document.querySelectorAll('.ecriture-check');
const selectAll = document.getElementById('selectAll');
const totalDebitEl = document.getElementById('total-debit');
const totalCreditEl = document.getElementById('total-credit');
const differenceEl = document.getElementById('difference');
const btnLettrer = document.getElementById('btn-lettrer');

function updateTotals() {
    let totalDebit = 0;
    let totalCredit = 0;
    
    ecritureChecks.forEach(check => {
        if (check.checked) {
            totalDebit += parseFloat(check.dataset.debit || 0);
            totalCredit += parseFloat(check.dataset.credit || 0);
        }
    });
    
    const diff = totalDebit - totalCredit;
    
    totalDebitEl.textContent = totalDebit.toFixed(2);
    totalCreditEl.textContent = totalCredit.toFixed(2);
    differenceEl.textContent = Math.abs(diff).toFixed(2);
    
    // Colorer la différence
    if (diff === 0 && totalDebit > 0) {
        differenceEl.className = 'f-w-600 text-success';
        btnLettrer.disabled = false;
    } else if (diff > 0) {
        differenceEl.className = 'f-w-600 text-danger';
        btnLettrer.disabled = true;
    } else if (diff < 0) {
        differenceEl.className = 'f-w-600 text-warning';
        btnLettrer.disabled = true;
    } else {
        differenceEl.className = 'f-w-600 text-muted';
        btnLettrer.disabled = true;
    }
}

// Select all
if (selectAll) {
    selectAll.addEventListener('change', function() {
        ecritureChecks.forEach(check => {
            check.checked = this.checked;
        });
        updateTotals();
    });
}

// Individual checks
ecritureChecks.forEach(check => {
    check.addEventListener('change', updateTotals);
});

// Initial update
updateTotals();
</script>
{% endblock %}
